public class Solution {
    public int countPrimes(int n) {
        int count=0;
        boolean [] a=new boolean[n];
       
        for(int i=2; i*i<n; i++) {  
        if(!a[i]) {  
            for(int j=i; i*j<n; j++) {  
                a[i*j] = true;  
            }  
        }  
     }  

     for(int i=2; i<n; i++) {  
         if(a[i] == false) ++count;  
     }  
     return count;  
    }
}
